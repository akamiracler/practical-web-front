### 1. let, const and Block Scoping

```js
console.log(productId);
var productId = 12;
// undefined
```
```js
console.log(productId);
let productId = 12;
// ReferenceError: productId is not defined
```
```js
let productId = 12;
console.log(productId); // 12
```
```js
let productId;
console.log(productId); // undefined
```
```js
let productId = 12; {
  let productId = 2000;
}
console.log(productId); // 12
```
```js
{
  let productId = 2000;
}
console.log(productId); // Reference Error: productId is not defined
```
```js
function updateProductId() {
  productId = 12;
}
let productId = null;
updateProductId();
console.log(productId); // 12
```
```js
function updateProductId() {
  productId = 12;
}
let productId = null;
updateProductId();
console.log(productId); // 12
```
```js
let productId = 42;
for (let productId = 0; productId < 10; productId++) {}
console.log(productId); // 42
```
```js
let updateFunctions = [];
for (var i = 0; i < 2; i++) {
  updateFunctions.push(function() {
    return i;
  });
}
console.log(updateFunctions[0]()); // 2
```
```js
let updateFunctions = [];
for (let i = 0; i < 2; i++) {
  updateFunctions.push(function() {
    return i;
  });
}
console.log(updateFunctions[0]()); // 0
```
```js
const MARKUP_PCT = 100;
console.log(MARKUP_PCT); // 100
```
```js
const MARKUP_PCT;
console.log(MARKUP_PCT); // SyntaxError: Unexpected token ;
```
```js
const MARKUP_PCT = 100;
MARKUP_PCT = 10;
console.log(MARKUP_PCT); // TypeError: Assignment to constant variable.
```
```js
const MARKUP_PCT = 100;
if (MARKUP_PCT > 0) {
  const MARKUP_PCT = 10;
}
console.log(MARKUP_PCT); // 100
```
