### 12.6. Math Extensions

```js
console.log(Math.sign(0)); // 0
console.log(Math.sign(-0)); // -0
console.log(Math.sign(-20)); // -1
console.log(Math.sign(20)); // 1
console.log(Math.sign(NaN)); // NaN
```
```js
console.log(Math.cbrt(27)); // 3
```
```js
console.log(Math.trunc(27.1)); // 27
console.log(Math.trunc(-27.9)); // -27
```
